pdf2png () {
    dpi=150
    if [ ${DPI} ]
    then
        dpi=${DPI}
    fi
    for filename in $@
    do
        pdftoppm -r ${dpi} -png -cropbox ${filename} ${filename/%.pdf/}
    done
}
